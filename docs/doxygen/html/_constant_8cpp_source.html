67 void Constant::Rotate(bool clockwise)
68 {
69  if(clockwise)
70  m_angle += 90.0;
71  else
72  m_angle -= 90.0;
73  if(m_angle >= 360.0)
74  m_angle = 0.0;
75  else if(m_angle < 0)
76  m_angle = 270.0;
77 
78  UpdatePoints();
79 
80  for(auto it = m_nodeList.begin(), itEnd = m_nodeList.end(); it != itEnd; ++it) {
81  Node* node = *it;
82  node->Rotate(clockwise);
83  }
84 }

86 void Constant::UpdatePoints()

95  } else if(m_angle == 270.0) {
